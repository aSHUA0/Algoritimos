#include<stdio.h>
#include<math.h>

main(){
	
	double numero, raiz, quad;
	
	printf("Escreva um numero qualquer\n");
	scanf("%lf", &numero);
	
	if (numero >= 0){
		raiz = sqrt(numero);
		printf("A raiz de %lf é %lf", numero, raiz);
		
	}else {
		quad = pow(numero,2);
		printf("O seu numero elevado ao quadrado é %.2lf", quad);
		
	}
	
}
